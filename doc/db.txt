equipmentio
{
    _id:        ObjectID,
    project:    ObjectID,
    type:       String, //checkout, checkin
    draft:      Boolean,
    reservation:ObjectID, //only available when it's a draft!
    time:       Number, // only present when not a draft
    user:       ObjectID, // only present when not a draft
    person:     ObjectID, //person picking up or receiving / returning equipment
    delivered:  Boolean, //out: true when deliviered, false when picked up by the customer, in: true when picked up by us, false when returned by the customer
    items:{
        TYPE:{
            count:  Number,
            suppliers:{
                SUPPLIER:{
                    count: Number,
                    ids:[String] //optional
                }
            }
        }
    },
    history:[
        {//ENTRY
            supplier: "SUPPLIER", //required
            type: TYPE, //required
            count: //required
            items: { //required for types with hasItems:true and supplier:'own'
                ITEM: [
                    ENTRY
                ]
            }
        }
    ]
}
